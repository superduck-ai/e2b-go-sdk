# Cloud buckets

This repository does not expose a first-party bucket client. To use external buckets from a sandbox, install a FUSE-based mount tool in a custom template and run the mount command inside the sandbox.

If you do not need an external bucket, use [Volumes](/docs/volumes) instead.

## Google Cloud Storage with `gcsfuse`

### Build a template

```go
package main

import e2b "github.com/superduck-ai/e2b-go-sdk"

var gcsTemplate = e2b.Template(nil).
	FromBaseImage().
	AptInstall([]string{"gnupg", "lsb-release"}, &struct{ User string }{User: "root"}).
	RunCmd("lsb_release -c -s > /tmp/lsb_release", &struct{ User string }{User: "root"}).
	RunCmd(`GCSFUSE_REPO=$(cat /tmp/lsb_release) && echo "deb [signed-by=/usr/share/keyrings/cloud.google.asc] https://packages.cloud.google.com/apt gcsfuse-$GCSFUSE_REPO main" > /etc/apt/sources.list.d/gcsfuse.list`, &struct{ User string }{User: "root"}).
	RunCmd("curl https://packages.cloud.google.com/apt/doc/apt-key.gpg > /usr/share/keyrings/cloud.google.asc", &struct{ User string }{User: "root"}).
	AptInstall([]string{"gcsfuse"}, &struct{ User string }{User: "root"})
```

### Mount the bucket at runtime

```go
package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "<your-template-id>", nil)
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	_, err = sandbox.Files.MakeDir(ctx, "/home/user/bucket", nil)
	if err != nil {
		log.Fatal(err)
	}
	_, err = sandbox.Files.Write(ctx, "/home/user/key.json", "<your service account key>", nil)
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Commands.Run(
		ctx,
		`gcsfuse -o allow_other --file-mode=777 --dir-mode=777 --key-file /home/user/key.json <bucket-name> /home/user/bucket`,
		&e2b.CommandStartOpts{
			User: "root",
		},
	)
	if err != nil {
		log.Fatal(err)
	}
}
```

## Amazon S3 with `s3fs`

### Build a template

```go
package main

import e2b "github.com/superduck-ai/e2b-go-sdk"

var s3Template = e2b.Template(nil).
	FromUbuntuImage("latest").
	AptInstall([]string{"s3fs"}, &struct{ User string }{User: "root"})
```

### Mount the bucket at runtime

```go
package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "<your-template-id>", nil)
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	_, err = sandbox.Files.MakeDir(ctx, "/home/user/bucket", nil)
	if err != nil {
		log.Fatal(err)
	}
	_, err = sandbox.Files.Write(ctx, "/home/user/.passwd-s3fs", "<AWS_ACCESS_KEY_ID>:<AWS_SECRET_ACCESS_KEY>", nil)
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Commands.Run(ctx, "chmod 600 /home/user/.passwd-s3fs", nil)
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Commands.Run(
		ctx,
		`sudo s3fs -o passwd_file=/home/user/.passwd-s3fs -o allow_other <bucket-name> /home/user/bucket`,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
}
```

## Cloudflare R2

R2 can use the same `s3fs` template. The runtime mount differs only by the endpoint URL.

```go
package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "<your-template-id>", nil)
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	_, err = sandbox.Files.MakeDir(ctx, "/home/user/bucket", nil)
	if err != nil {
		log.Fatal(err)
	}
	_, err = sandbox.Files.Write(ctx, "/home/user/.passwd-s3fs", "<R2_ACCESS_KEY_ID>:<R2_SECRET_ACCESS_KEY>", nil)
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Commands.Run(ctx, "chmod 600 /home/user/.passwd-s3fs", nil)
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Commands.Run(
		ctx,
		`sudo s3fs -o passwd_file=/home/user/.passwd-s3fs -o url=https://<ACCOUNT_ID>.r2.cloudflarestorage.com -o allow_other <bucket-name> /home/user/bucket`,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
}
```

Adjust the mount flags for your provider, region, and permission model. The important Go-side pattern is unchanged: install the FUSE tool in the template, then mount the bucket during sandbox runtime.
