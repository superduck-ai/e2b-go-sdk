# Static charts

For static charts, save the image to a file inside the sandbox and then read it back through `sandbox.Files.Read(...)`.

```go
package main

import (
	"context"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "python-data", nil)
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	execution, err := sandbox.Commands.Run(ctx, `python3 - <<'PY'
import matplotlib.pyplot as plt

plt.plot([1, 2, 3, 4])
plt.ylabel("some numbers")
plt.savefig("/home/user/chart.png")
PY`, nil)
	if err != nil {
		log.Fatal(err)
	}
	log.Println(execution.(*e2b.CommandResult).Stdout)

	value, err := sandbox.Files.Read(ctx, "/home/user/chart.png", &e2b.FilesystemReadOpts{
		Format: e2b.ReadFormatBytes,
	})
	if err != nil {
		log.Fatal(err)
	}

	if err := os.WriteFile("chart.png", value.([]byte), 0o644); err != nil {
		log.Fatal(err)
	}
}
```
