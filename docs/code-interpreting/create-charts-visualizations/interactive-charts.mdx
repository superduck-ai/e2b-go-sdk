# Interactive charts

This Go SDK does not return chart objects automatically.

The usual pattern is to write chart data to JSON inside the sandbox, then read that JSON back and render it with your frontend charting library.

```go
package main

import (
	"context"
	"encoding/json"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

type ChartElement struct {
	Label string  `json:"label"`
	Value float64 `json:"value"`
	Group string  `json:"group"`
}

type ChartPayload struct {
	Type     string         `json:"type"`
	Title    string         `json:"title"`
	XLabel   string         `json:"x_label"`
	YLabel   string         `json:"y_label"`
	Elements []ChartElement `json:"elements"`
}

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "python-data", nil)
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	_, err = sandbox.Files.Write(ctx, "/home/user/chart.json", `{
  "type": "bar",
  "title": "Book Sales by Authors",
  "x_label": "Authors",
  "y_label": "Books Sold",
  "elements": [
    {"label": "Author A", "value": 100, "group": "Books Sold"},
    {"label": "Author B", "value": 200, "group": "Books Sold"}
  ]
}`, nil)
	if err != nil {
		log.Fatal(err)
	}

	value, err := sandbox.Files.Read(ctx, "/home/user/chart.json", nil)
	if err != nil {
		log.Fatal(err)
	}

	var payload ChartPayload
	if err := json.Unmarshal([]byte(value.(string)), &payload); err != nil {
		log.Fatal(err)
	}

	log.Println(payload.Type, payload.Title, len(payload.Elements))
}
```

Send the parsed JSON to your frontend and render it with your own chart library.
