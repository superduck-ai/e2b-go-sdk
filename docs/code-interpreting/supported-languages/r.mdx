# Run R code

Build or choose a template with `Rscript` available, then execute R through the normal command API.

```go
package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

var template = e2b.Template(nil).
	FromUbuntuImage("24.04").
	AptInstall([]string{"r-base"}, &struct{ User string }{User: "root"})

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "r-runtime", nil)
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	execution, err := sandbox.Commands.Run(ctx, `Rscript -e 'print("Hello, world!")'`, nil)
	if err != nil {
		log.Fatal(err)
	}

	log.Println(execution.(*e2b.CommandResult).Stdout)
}
```
