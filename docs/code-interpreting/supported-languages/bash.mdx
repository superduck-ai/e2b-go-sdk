# Run bash code

For Bash, run shell commands directly with `sandbox.Commands.Run(...)` or use a PTY for interactive sessions.

```go
package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "base", nil)
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	execution, err := sandbox.Commands.Run(ctx, `bash -lc 'echo "Hello, world!"'`, nil)
	if err != nil {
		log.Fatal(err)
	}

	log.Println(execution.(*e2b.CommandResult).Stdout)
}
```
