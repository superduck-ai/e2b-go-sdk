# Claude Code

Claude Code can run inside an E2B sandbox with filesystem, terminal, Git, and optional MCP access.

If your E2B account has the prebuilt `claude` template, you can create it directly and invoke the CLI from Go.

## Run headless

```go
package main

import (
	"context"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "claude", &e2b.SandboxOpts{
		Envs: map[string]string{
			"ANTHROPIC_API_KEY": os.Getenv("ANTHROPIC_API_KEY"),
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	execution, err := sandbox.Commands.Run(
		ctx,
		`claude --dangerously-skip-permissions -p "Create a hello world HTTP server in Go"`,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	result := execution.(*e2b.CommandResult)

	log.Println(result.Stdout)
}
```

## Work on a cloned repository

```go
package main

import (
	"context"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()
	timeoutMs := 600_000
	depth := 1

	sandbox, err := e2b.Create(ctx, "claude", &e2b.SandboxOpts{
		Envs: map[string]string{
			"ANTHROPIC_API_KEY": os.Getenv("ANTHROPIC_API_KEY"),
		},
		TimeoutMs: &timeoutMs,
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	_, err = sandbox.Git.Clone(ctx, "https://github.com/your-org/your-repo.git", &e2b.GitCloneOpts{
		Path:     "/home/user/repo",
		Username: "x-access-token",
		Password: os.Getenv("GITHUB_TOKEN"),
		Depth:    &depth,
	})
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Commands.Run(
		ctx,
		`cd /home/user/repo && claude --dangerously-skip-permissions -p "Add error handling to all API endpoints"`,
		&e2b.CommandStartOpts{
			OnStdout: func(data e2b.Stdout) {
				log.Print(data)
			},
		},
	)
	if err != nil {
		log.Fatal(err)
	}

	diffExecution, err := sandbox.Commands.Run(ctx, "cd /home/user/repo && git diff", nil)
	if err != nil {
		log.Fatal(err)
	}
	diff := diffExecution.(*e2b.CommandResult)

	log.Println(diff.Stdout)
}
```

## Structured output

```go
package main

import (
	"context"
	"encoding/json"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"

	sandbox, err := e2b.Create(ctx, "claude", &e2b.SandboxOpts{
		Envs: map[string]string{
			"ANTHROPIC_API_KEY": os.Getenv("ANTHROPIC_API_KEY"),
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	execution, err := sandbox.Commands.Run(
		ctx,
		`claude --dangerously-skip-permissions --output-format json -p "Review this codebase and list all security issues as JSON"`,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	result := execution.(*e2b.CommandResult)

	var response map[string]any
	if err := json.Unmarshal([]byte(result.Stdout), &response); err != nil {
		log.Fatal(err)
	}

	log.Println(response)
}
```

## Streaming output

```go
package main

import (
	"context"
	"encoding/json"
	"log"
	"os"
	"strings"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "claude", &e2b.SandboxOpts{
		Envs: map[string]string{
			"ANTHROPIC_API_KEY": os.Getenv("ANTHROPIC_API_KEY"),
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	_, err = sandbox.Commands.Run(
		ctx,
		`cd /home/user/repo && claude --dangerously-skip-permissions --output-format stream-json -p "Find and fix all TODO comments"`,
		&e2b.CommandStartOpts{
			OnStdout: func(data e2b.Stdout) {
				for _, line := range strings.Split(string(data), "\n") {
					if strings.TrimSpace(line) == "" {
						continue
					}
					var event map[string]any
					_ = json.Unmarshal([]byte(line), &event)
					log.Println(event["type"])
				}
			},
		},
	)
	if err != nil {
		log.Fatal(err)
	}
}
```

## Resume a session

```go
package main

import (
	"context"
	"encoding/json"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()
	timeoutMs := 600_000

	sandbox, err := e2b.Create(ctx, "claude", &e2b.SandboxOpts{
		Envs: map[string]string{
			"ANTHROPIC_API_KEY": os.Getenv("ANTHROPIC_API_KEY"),
		},
		TimeoutMs: &timeoutMs,
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	initialExecution, err := sandbox.Commands.Run(
		ctx,
		`cd /home/user/repo && claude --dangerously-skip-permissions --output-format json -p "Analyze the codebase and create a refactoring plan"`,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	initial := initialExecution.(*e2b.CommandResult)

	var response map[string]any
	if err := json.Unmarshal([]byte(initial.Stdout), &response); err != nil {
		log.Fatal(err)
	}
	sessionID, _ := response["session_id"].(string)

	_, err = sandbox.Commands.Run(
		ctx,
		`cd /home/user/repo && claude --dangerously-skip-permissions --resume `+sessionID+` -p "Now implement step 1 of the plan"`,
		&e2b.CommandStartOpts{
			OnStdout: func(data e2b.Stdout) {
				log.Print(data)
			},
		},
	)
	if err != nil {
		log.Fatal(err)
	}
}
```

## Connect MCP tools

```go
package main

import (
	"context"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "claude", &e2b.SandboxOpts{
		Envs: map[string]string{
			"ANTHROPIC_API_KEY": os.Getenv("ANTHROPIC_API_KEY"),
		},
		Mcp: e2b.McpServer{
			"browserbase": map[string]any{
				"apiKey":    os.Getenv("BROWSERBASE_API_KEY"),
				"projectId": os.Getenv("BROWSERBASE_PROJECT_ID"),
			},
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	mcpURL := sandbox.GetMcpUrl()
	mcpToken, tokenErr := sandbox.GetMcpToken()
	if tokenErr != nil {
		log.Fatal(tokenErr)
	}

	_, err = sandbox.Commands.Run(
		ctx,
		`claude mcp add --transport http e2b-mcp-gateway `+mcpURL+` --header "Authorization: Bearer `+mcpToken+`"`,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Commands.Run(
		ctx,
		`claude --dangerously-skip-permissions -p "Use browserbase to research E2B and summarize your findings"`,
		&e2b.CommandStartOpts{
			OnStdout: func(data e2b.Stdout) {
				log.Print(data)
			},
		},
	)
	if err != nil {
		log.Fatal(err)
	}
}
```

## Build a custom Claude template

```go
package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	template := e2b.Template(nil).FromTemplate("claude")

	_, err := e2b.Build(context.Background(), template, "my-claude", &e2b.BuildOptions{
		BasicBuildOptions: e2b.BasicBuildOptions{
			CpuCount:    2,
			MemoryMB:    2048,
			OnBuildLogs: e2b.DefaultBuildLogger(),
		},
	})
	if err != nil {
		log.Fatal(err)
	}
}
```
