# OpenClaw

OpenClaw can run inside a regular E2B sandbox with terminal, filesystem, and Git access.

If your E2B account has a prebuilt `openclaw` template, create a sandbox from that template directly and invoke the OpenClaw CLI with `sandbox.Commands.Run(...)`.

## Run headless

```go
package main

import (
	"context"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "openclaw", &e2b.SandboxOpts{
		Envs: map[string]string{
			"ANTHROPIC_API_KEY": os.Getenv("ANTHROPIC_API_KEY"),
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	execution, err := sandbox.Commands.Run(
		ctx,
		`openclaw agent --local --thinking high --message "Create a hello world HTTP server in Go"`,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	result := execution.(*e2b.CommandResult)

	log.Println(result.Stdout)
}
```

## Work on a cloned repository

```go
package main

import (
	"context"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()
	timeoutMs := 600_000
	depth := 1

	sandbox, err := e2b.Create(ctx, "openclaw", &e2b.SandboxOpts{
		Envs: map[string]string{
			"ANTHROPIC_API_KEY": os.Getenv("ANTHROPIC_API_KEY"),
		},
		TimeoutMs: &timeoutMs,
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	_, err = sandbox.Git.Clone(ctx, "https://github.com/your-org/your-repo.git", &e2b.GitCloneOpts{
		Path:     "/home/user/repo",
		Username: "x-access-token",
		Password: os.Getenv("GITHUB_TOKEN"),
		Depth:    &depth,
	})
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Commands.Run(
		ctx,
		`cd /home/user/repo && openclaw agent --local --thinking high --message "Add error handling to all API endpoints"`,
		&e2b.CommandStartOpts{
			OnStdout: func(data e2b.Stdout) {
				log.Print(data)
			},
		},
	)
	if err != nil {
		log.Fatal(err)
	}

	diffExecution, err := sandbox.Commands.Run(ctx, "cd /home/user/repo && git diff", nil)
	if err != nil {
		log.Fatal(err)
	}
	diff := diffExecution.(*e2b.CommandResult)

	log.Println(diff.Stdout)
}
```

## Structured output

Use `--json` when you want a machine-readable response.

```go
package main

import (
	"context"
	"encoding/json"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "openclaw", &e2b.SandboxOpts{
		Envs: map[string]string{
			"ANTHROPIC_API_KEY": os.Getenv("ANTHROPIC_API_KEY"),
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	execution, err := sandbox.Commands.Run(
		ctx,
		`openclaw agent --local --json --message "List all files in the current directory and describe each"`,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	result := execution.(*e2b.CommandResult)

	var response map[string]any
	if err := json.Unmarshal([]byte(result.Stdout), &response); err != nil {
		log.Fatal(err)
	}

	log.Println(response)
}
```

## Customize `SOUL.md`

Write `SOUL.md` into the default workspace before starting the agent.

```go
package main

import (
	"context"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "openclaw", &e2b.SandboxOpts{
		Envs: map[string]string{
			"ANTHROPIC_API_KEY": os.Getenv("ANTHROPIC_API_KEY"),
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	soul := `# Soul

## Core Truths
- Be genuinely helpful.
- Read the workspace before asking for more context.

## Boundaries
- Do not leak user data.
- Be cautious with destructive actions.
`

	_, err = sandbox.Files.Write(ctx, "/home/user/.openclaw/workspaces/default/SOUL.md", soul, nil)
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Commands.Run(
		ctx,
		`openclaw agent --local --thinking high --message "Introduce yourself"`,
		&e2b.CommandStartOpts{
			OnStdout: func(data e2b.Stdout) {
				log.Print(data)
			},
		},
	)
	if err != nil {
		log.Fatal(err)
	}
}
```

## Build a custom template

If you need extra packages or want the gateway to start automatically, build your own template on top of `openclaw`.

```go
package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	template := e2b.Template(nil).
		FromTemplate("openclaw").
		SetStartCmd(
			"openclaw gateway --port 18789 --verbose",
			e2b.WaitForPort(18789),
		)

	buildInfo, err := e2b.Build(context.Background(), template, "my-openclaw", &e2b.BuildOptions{
		BasicBuildOptions: e2b.BasicBuildOptions{
			CpuCount:    2,
			MemoryMB:    2048,
			OnBuildLogs: e2b.DefaultBuildLogger(),
		},
	})
	if err != nil {
		log.Fatal(err)
	}

	log.Println(buildInfo.TemplateID)
}
```
