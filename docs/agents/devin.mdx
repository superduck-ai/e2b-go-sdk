# Devin

Devin for Terminal can run in an E2B sandbox as an isolated coding workspace.

Installation and login are interactive, so use `sandbox.Pty` for setup. After that, reconnect to the same sandbox and use `sandbox.Commands.Run(...)` for non-interactive tasks.

## Install and log in with a PTY

```go
package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()
	sandboxTimeoutMs := 3600_000
	ptyTimeoutMs := 0

	sandbox, err := e2b.Create(ctx, "base", &e2b.SandboxOpts{
		TimeoutMs: &sandboxTimeoutMs,
		Lifecycle: &e2b.SandboxLifecycle{
			OnTimeout: "pause",
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	terminal, err := sandbox.Pty.Create(ctx, &e2b.PtyCreateOpts{
		Cols:      120,
		Rows:      36,
		TimeoutMs: &ptyTimeoutMs,
		OnData: func(data e2b.PtyOutput) {
			log.Print(string(data))
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	defer terminal.Kill()

	if err := sandbox.Pty.SendInput(ctx, terminal.Pid, []byte("curl -fsSL https://cli.devin.ai/install.sh | bash\n"), nil); err != nil {
		log.Fatal(err)
	}
	if err := sandbox.Pty.SendInput(ctx, terminal.Pid, []byte("source /home/user/.bashrc\n"), nil); err != nil {
		log.Fatal(err)
	}
	if err := sandbox.Pty.SendInput(ctx, terminal.Pid, []byte("devin login\n"), nil); err != nil {
		log.Fatal(err)
	}

	log.Println("sandbox id:", sandbox.SandboxID)
}
```

Keep that sandbox running or paused after login so Devin keeps its authenticated state.

## Run a non-interactive prompt

Reconnect to the same sandbox after installation and login, then use `devin -p` for headless tasks.

```go
package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Connect(ctx, "sbx_123", nil)
	if err != nil {
		log.Fatal(err)
	}

	execution, err := sandbox.Commands.Run(
		ctx,
		`mkdir -p /home/user/project && cd /home/user/project && devin --permission-mode dangerous -p "Create a hello world HTTP server in Go"`,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	result := execution.(*e2b.CommandResult)

	log.Println(result.Stdout)
}
```

## Work on a cloned repository

```go
package main

import (
	"context"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()
	depth := 1

	sandbox, err := e2b.Connect(ctx, "sbx_123", nil)
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Git.Clone(ctx, "https://github.com/your-org/your-repo.git", &e2b.GitCloneOpts{
		Path:     "/home/user/repo",
		Username: "x-access-token",
		Password: os.Getenv("GITHUB_TOKEN"),
		Depth:    &depth,
	})
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Commands.Run(
		ctx,
		`cd /home/user/repo && devin --permission-mode dangerous -p "Add error handling to all API endpoints"`,
		&e2b.CommandStartOpts{
			OnStdout: func(data e2b.Stdout) {
				log.Print(data)
			},
		},
	)
	if err != nil {
		log.Fatal(err)
	}

	diffExecution, err := sandbox.Commands.Run(ctx, "cd /home/user/repo && git diff", nil)
	if err != nil {
		log.Fatal(err)
	}
	diff := diffExecution.(*e2b.CommandResult)

	log.Println(diff.Stdout)
}
```
