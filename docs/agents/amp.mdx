# Amp

Amp can run inside an E2B sandbox with full terminal, filesystem, and Git access.

If your E2B account has the prebuilt `amp` template, create a sandbox from it and invoke the Amp CLI through `sandbox.Commands.Run(...)`.

## Run headless

```go
package main

import (
	"context"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "amp", &e2b.SandboxOpts{
		Envs: map[string]string{
			"AMP_API_KEY": os.Getenv("AMP_API_KEY"),
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	execution, err := sandbox.Commands.Run(
		ctx,
		`amp --dangerously-allow-all -x "Create a hello world HTTP server in Go"`,
		nil,
	)
	if err != nil {
		log.Fatal(err)
	}
	result := execution.(*e2b.CommandResult)

	log.Println(result.Stdout)
}
```

## Work on a cloned repository

```go
package main

import (
	"context"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()
	timeoutMs := 600_000
	depth := 1

	sandbox, err := e2b.Create(ctx, "amp", &e2b.SandboxOpts{
		Envs: map[string]string{
			"AMP_API_KEY": os.Getenv("AMP_API_KEY"),
		},
		TimeoutMs: &timeoutMs,
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	_, err = sandbox.Git.Clone(ctx, "https://github.com/your-org/your-repo.git", &e2b.GitCloneOpts{
		Path:     "/home/user/repo",
		Username: "x-access-token",
		Password: os.Getenv("GITHUB_TOKEN"),
		Depth:    &depth,
	})
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Commands.Run(
		ctx,
		`cd /home/user/repo && amp --dangerously-allow-all -x "Add error handling to all API endpoints"`,
		&e2b.CommandStartOpts{
			OnStdout: func(data e2b.Stdout) {
				log.Print(data)
			},
		},
	)
	if err != nil {
		log.Fatal(err)
	}

	diffExecution, err := sandbox.Commands.Run(ctx, "cd /home/user/repo && git diff", nil)
	if err != nil {
		log.Fatal(err)
	}
	diff := diffExecution.(*e2b.CommandResult)

	log.Println(diff.Stdout)
}
```

## Streaming JSON

```go
package main

import (
	"context"
	"encoding/json"
	"log"
	"os"
	"strings"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	sandbox, err := e2b.Create(ctx, "amp", &e2b.SandboxOpts{
		Envs: map[string]string{
			"AMP_API_KEY": os.Getenv("AMP_API_KEY"),
		},
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	_, err = sandbox.Commands.Run(
		ctx,
		`cd /home/user/repo && amp --dangerously-allow-all --stream-json -x "Find and fix all TODO comments"`,
		&e2b.CommandStartOpts{
			OnStdout: func(data e2b.Stdout) {
				for _, line := range strings.Split(string(data), "\n") {
					if strings.TrimSpace(line) == "" {
						continue
					}
					var event map[string]any
					_ = json.Unmarshal([]byte(line), &event)
					log.Println(event["type"])
				}
			},
		},
	)
	if err != nil {
		log.Fatal(err)
	}
}
```

## Thread management

```go
package main

import (
	"context"
	"encoding/json"
	"log"
	"os"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()
	timeoutMs := 600_000

	sandbox, err := e2b.Create(ctx, "amp", &e2b.SandboxOpts{
		Envs: map[string]string{
			"AMP_API_KEY": os.Getenv("AMP_API_KEY"),
		},
		TimeoutMs: &timeoutMs,
	})
	if err != nil {
		log.Fatal(err)
	}
	defer sandbox.Kill(context.Background(), nil)

	_, err = sandbox.Commands.Run(
		ctx,
		`cd /home/user/repo && amp --dangerously-allow-all -x "Analyze the codebase and create a refactoring plan"`,
		&e2b.CommandStartOpts{
			OnStdout: func(data e2b.Stdout) {
				log.Print(data)
			},
		},
	)
	if err != nil {
		log.Fatal(err)
	}

	threadsExecution, err := sandbox.Commands.Run(ctx, "amp threads list --json", nil)
	if err != nil {
		log.Fatal(err)
	}
	threads := threadsExecution.(*e2b.CommandResult)

	var threadList []map[string]any
	if err := json.Unmarshal([]byte(threads.Stdout), &threadList); err != nil {
		log.Fatal(err)
	}
	threadID, _ := threadList[0]["id"].(string)

	_, err = sandbox.Commands.Run(
		ctx,
		`cd /home/user/repo && amp threads continue `+threadID+` --dangerously-allow-all -x "Now implement step 1 of the plan"`,
		&e2b.CommandStartOpts{
			OnStdout: func(data e2b.Stdout) {
				log.Print(data)
			},
		},
	)
	if err != nil {
		log.Fatal(err)
	}
}
```

## Build a custom Amp template

```go
package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	template := e2b.Template(nil).FromTemplate("amp")

	_, err := e2b.Build(context.Background(), template, "my-amp", &e2b.BuildOptions{
		BasicBuildOptions: e2b.BasicBuildOptions{
			CpuCount:    2,
			MemoryMB:    2048,
			OnBuildLogs: e2b.DefaultBuildLogger(),
		},
	})
	if err != nil {
		log.Fatal(err)
	}
}
```
