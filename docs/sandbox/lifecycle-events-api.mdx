# Monitor sandbox lifecycle events

The Go SDK does not currently provide dedicated helpers for lifecycle-events endpoints. Use the control-plane API directly with `net/http` and your E2B API key.

Available query parameters:

- `offset`: number of events to skip
- `limit`: number of events to return
- `orderAsc`: set to `true` for ascending order
- `types`: repeat this parameter to filter by one or more event types

## Fetch events

```go
package main

import (
	"context"
	"encoding/json"
	"fmt"
	"log"
	"net/http"
	"net/url"
	"os"
)

func fetchEvents(ctx context.Context, client *http.Client, apiKey, endpoint string) ([]map[string]any, error) {
	req, err := http.NewRequestWithContext(ctx, http.MethodGet, endpoint, nil)
	if err != nil {
		return nil, err
	}
	req.Header.Set("X-API-Key", apiKey)

	resp, err := client.Do(req)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		return nil, fmt.Errorf("unexpected status: %s", resp.Status)
	}

	var events []map[string]any
	if err := json.NewDecoder(resp.Body).Decode(&events); err != nil {
		return nil, err
	}

	return events, nil
}

func main() {
	ctx := context.Background()
	client := &http.Client{}
	apiKey := os.Getenv("E2B_API_KEY")
	sandboxID := "sbx_123"

	sandboxEvents, err := fetchEvents(
		ctx,
		client,
		apiKey,
		"https://api.e2b.app/events/sandboxes/"+sandboxID,
	)
	if err != nil {
		log.Fatal(err)
	}

	teamEvents, err := fetchEvents(
		ctx,
		client,
		apiKey,
		"https://api.e2b.app/events/sandboxes?limit=10",
	)
	if err != nil {
		log.Fatal(err)
	}

	query := url.Values{}
	query.Add("types", "sandbox.lifecycle.created")
	query.Add("types", "sandbox.lifecycle.killed")

	filteredEvents, err := fetchEvents(
		ctx,
		client,
		apiKey,
		"https://api.e2b.app/events/sandboxes?"+query.Encode(),
	)
	if err != nil {
		log.Fatal(err)
	}

	fmt.Println(len(sandboxEvents), len(teamEvents), len(filteredEvents))
}
```

The decoded event payloads include IDs, event type, timestamp, sandbox identifiers, and optional event-specific metadata.
