# Mounting Volumes

Mount volumes through `SandboxOpts.VolumeMounts`.

In Go, `VolumeMounts` is a `map[string]any` where:

- the key is the mount path inside the sandbox
- the value can be a volume name string
- or a value with a `Name` field, including `*e2b.Volume`

## Mount a volume object

```go
package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"
)

func main() {
	ctx := context.Background()

	volume, err := e2b.CreateVolume(ctx, "my-volume", nil)
	if err != nil {
		log.Fatal(err)
	}

	sandbox, err := e2b.Create(ctx, "", &e2b.SandboxOpts{
		VolumeMounts: map[string]any{
			"/mnt/my-data": volume,
		},
	})
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Files.Write(ctx, "/mnt/my-data/hello.txt", "Hello, world!", nil)
	if err != nil {
		log.Fatal(err)
	}
}
```

package main

import (
	"context"
	"log"

	e2b "github.com/superduck-ai/e2b-go-sdk"

		},
	})
	if err != nil {
		log.Fatal(err)
	}

	_, err = sandbox.Files.Write(ctx, "/mnt/my-data/hello.txt", "Hello, world!", nil)

	}
